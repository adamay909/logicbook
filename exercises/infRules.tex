 \item Each of the following little derivation has a mistake. What is it?

  \begin{enumerate}

   \item    
%title: %
\begin{argumentN}[1]
%generated  by  gentzen

\ai{p}{q}{A}

\end{argumentN}

\opts{
\dotline
\dotline
}{The datum and succedent cannot differ when using \p{A}}

\item 

%title: %
\begin{argumentN}[1]
%generated  by  gentzen

\ai{\Gamma }{p\mc{\lor  }q}{premise}

\ai{\Gamma }{q}{1,\disjE}

\end{argumentN}

\opts{
\dotline
\dotline
}{\disjE{} cannot be used in that way.}

\item 
%title: %
\begin{argumentN}[1]
%generated  by  gentzen

\ai{p}{p}{A}

\ai{q}{q}{A}

\ai{p\mc{\land  }q}{p\mc{\land  }q}{1,2,\conjI}

\end{argumentN}

\opts{

 \dotline
 \dotline
}
{\conjI{} does not result in that datum.}

\item 
%title: %
\begin{argumentN}[1]
%generated  by  gentzen

\ai{\Gamma ,  s}{r}{premise}

\ai{\Gamma ,  q}{p}{premise}

\ai{\Gamma ,  s,  q}{r\mc{\limplies  }p}{1,2,\conjI}

\end{argumentN}

\opts{
 \dotline
 \dotline
}
{\conjI{} results in a succedent with \p{\land} as main connective.}

\item 
%title: %
\begin{argumentN}[1]
%generated  by  gentzen

\ai{s}{s}{A}

\ai{s\mc{\lor  }w}{s}{1,\disjI}

\end{argumentN}

\opts{
 \dotline
 \dotline
}
{\disjI{} does not change datum.}

\item 
%title: %
\begin{argumentN}[1]
%generated  by  gentzen

\ai{p\mc{\land  }q}{p\mc{\land  }q}{A}

\ai{p}{p\mc{\land  }q}{1,\conjE}

\end{argumentN}

\opts{

 \dotline
 \dotline
}
{
Cannot eliminate a conjunct from a conjunction in datum using \conjE{}.}

\item 
%title: %
\begin{argumentN}[1]
%generated  by  gentzen

\ai{r}{s\mc{\lor  }w}{premise}

\ai{r}{s\mc{\lor  }(w\lor  q)}{1,\disjI}

\end{argumentN}

\opts{

 \dotline
 \dotline
}
{Cannot introduce a disjunct in that way using \disjI.}

\item 
%title: %
\begin{argumentN}[1]
%generated  by  gentzen

\ai{\Gamma ,  q}{p}{premise}

\ai{\Delta }{\mc{\lnot  }p}{premise}

\ai{\Gamma ,  \Delta }{\mc{\lnot  }q}{1,2,\negI}

\end{argumentN}


\opts{

 \dotline
 \dotline
}
{\p{q} is not in the datum of line 2.}

\item 
%title: %
\begin{argumentN}[1]
%generated  by  gentzen

\ai{\Gamma }{w}{premise}

\ai{\Gamma }{\mc{\lnot  }\lnot  w}{1,\negI}

\end{argumentN}


\opts{

 \dotline
 \dotline
}
{Cannot introduce double negation using \negI.}

\item 
%title: %
\begin{argumentN}[1]
%generated  by  gentzen

\ai{\Delta }{\lnot  \lnot  p\mc{\lor  }r}{premise}

\ai{\Delta }{p\mc{\lor  }r}{1,\negE}

\end{argumentN}

\opts{

 \dotline
 \dotline
}
{Can't eliminate the double negation in this way.}

\item

%title: %
\begin{argumentN}[1]
%generated  by  gentzen

\ai{p\mc{\limplies  }q}{p\mc{\limplies  }q}{A}

\ai{q}{q}{A}

\ai{p\mc{\limplies  }q,  q}{p}{1,2,\condE}

\end{argumentN}

\opts{

 \dotline
 \dotline
}
{Not how \condE{} works.}

\item
%title: %
\begin{argumentN}[1]
%generated  by  gentzen

\ai{p\mc{\limplies  }q}{p\mc{\limplies  }q}{A}

\ai{p\mc{\limplies  }q}{p}{1,\condE}

\end{argumentN}

\opts{

 \dotline
 \dotline
}
{Not how \condE{} works}

\item 
%title: %
\begin{argumentN}[1]
%generated  by  gentzen

\ai{\Gamma }{a}{premise}

\ai{\Delta }{b}{premise}

\ai{\Gamma ,  \Delta }{a\mc{\limplies  }b}{1,2,\condI}

\end{argumentN}

\opts{

 \dotline
 \dotline
}
{Not how \condI{} works.}

\end{enumerate}

\item The following have blanks that are impossible to fill in in accordance 
 with our inference rules if we stick to the annotations.  Explain why.

 \newcommand{\blanksp}{\_\_}

 \begin{enumerate}
	\setlength{\itemsep}{2em}

  \item 
%title: %
\begin{argumentN}[1]
%generated  by  gentzen

 \ai{\blanksp}{r}{A}

\ai{p,  q}{p}{1}

\end{argumentN}

\opts{

 \dotline
 \dotline

}{
 The blank would have to be an \p{r} given the annotation of line 1, but you 
can't get line 2 from that in the way indicated.}

\item 
%title: %
\begin{argumentN}[1]
%generated  by  gentzen

\ai{r}{r}{A}

\ai{\blanksp}{\blanksp}{1}

\ai{r}{p\mc{\limplies  }q}{2,\condI}

\end{argumentN}

\opts{

 \dotline
 \dotline

}
{
 Given the annotation of line 3, line 2 must have a \p{q} in succedent, but you 
can't get that given line 2 is said to be a rewrite of line 1.}

\item 
%title: %
\begin{argumentN}[1]
%generated  by  gentzen

\ai{\Gamma }{r}{premise}

\ai{s}{s}{A}

\ai{s}{\blanksp}{2,\disjI}

\ai{\Gamma ,  s}{r\mc{\land  }(s\land  q)}{1,3,\conjI}

\end{argumentN}

\opts{

 \dotline
 \dotline

}
{
 Given the annotation of line 4, the succedent of line 3 must be \p{s\land q}, 
 but you can't get that since the annotation of line 3 requires that the 
 succedent be a disjunction.
}

\item 
%title: %
\begin{argumentN}[1]
%generated  by  gentzen

\ai{\Gamma }{\mc{\lnot  }\lnot  p}{premise}

\ai{\blanksp}{\blanksp}{1,\negE}

\ai{\Delta }{p\mc{\limplies  }q}{premise}

\ai{\Delta }{q}{2,3,\condE}

\end{argumentN}

\opts{

 \dotline
 \dotline

}
{Given the annotation of line 4, the datum of line 2 must either be empty or 
\p{\Delta}. But the annotation of 2 requires that the datum be \p{\Gamma}.}
